Fight Game


Side server

while (true)
{
	receive("New Gamer", {
		send("Status", gamer.stats)
	})
	receive("Action", gamer.button, {
		CheckIfPossible(gamer.button)
		if (gamer.button == Success) {
			gamer.status.hp -= 1
			send(gamer.stats)
		}
		else {
			gamer.status.guard = -1
			send("ReloadStats", gamer.stats)
		}
	})
	receive("HpGamer", {
		if ("HpGamer" <= 0)
			send(gamer.stats, "KO")
	})
}
